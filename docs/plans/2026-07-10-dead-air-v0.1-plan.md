# Dead Air v0.1 (Test Room) Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** A playable single-room twin-stick shooter: gamepad/keyboard+mouse controls, three zombie waves, score, juice, win/lose screens.

**Architecture:** Plain JS loaded via classic `<script>` tags (no modules — the game must run from `file://` by double-clicking `index.html`). All code hangs off a single global namespace object `DA`. Fixed 1280×720 internal resolution scaled/letterboxed to the window. Entities are plain objects in arrays; circle-vs-circle collision. Pure logic (math, collision, wave sequencing) is unit-tested in a zero-dependency `tests.html`; rendering and feel are verified by playing.

**Tech Stack:** HTML5 canvas, vanilla JavaScript, Gamepad API. No npm, no build step, no framework.

**Design doc:** `docs/plans/2026-07-10-dead-air-design.md`

**Conventions for every task:**
- Script load order in `index.html` matters: `util.js`, `input.js`, `effects.js`, `bullets.js`, `enemies.js`, `player.js`, `rooms.js`, `main.js`. Add each file to `index.html` in the task that creates it.
- Every new pure function goes on `DA` and gets a test in `tests.html` BEFORE implementation (write test → open tests.html → see ❌ → implement → see ✅).
- Manual verification = open `index.html` in a browser (or the Claude browser preview) and check the listed behavior.
- Commit after every task with the message given.

```js
var DA = {
  W: 1280, H: 720,
  clamp: function (v, lo, hi) { return v < lo ? lo : (v > hi ? hi : v); },
  dist2: function (ax, ay, bx, by) { var dx = ax - bx, dy = ay - by; return dx * dx + dy * dy; },
  // circle-vs-circle hit test
  circleHit: function (ax, ay, ar, bx, by, br) {
    var r = ar + br; return DA.dist2(ax, ay, bx, by) <= r * r;
  },
  // normalize a vector; returns {x,y,len}. Zero-length stays zero.
  norm: function (x, y) {
    var len = Math.sqrt(x * x + y * y);
    if (len < 0.0001) return { x: 0, y: 0, len: 0 };
    return { x: x / len, y: y / len, len: len };
  },
  rand: function (lo, hi) { return lo + Math.random() * (hi - lo); }
};
```

**Step 3: Implement `js/input.js`**

```js
(function () {
  var DEAD = 0.2;
  DA.applyDeadzone = function (v) {
    var a = Math.abs(v);
    if (a < DEAD) return 0;
    return Math.sign(v) * (a - DEAD) / (1 - DEAD);
  };
  DA.screenToCanvas = function (sx, sy, winW, winH) {
    var scale = Math.min(winW / DA.W, winH / DA.H);
    var offX = (winW - DA.W * scale) / 2, offY = (winH - DA.H * scale) / 2;
    return { x: (sx - offX) / scale, y: (sy - offY) / scale };
  };

  var keys = {}, mouse = { x: DA.W / 2, y: DA.H / 2, down: false };
  var device = 'keyboard'; // or 'gamepad'

  window.addEventListener('keydown', function (e) { keys[e.code] = true; device = 'keyboard'; });
  window.addEventListener('keyup', function (e) { keys[e.code] = false; });
  window.addEventListener('mousemove', function (e) {
    var p = DA.screenToCanvas(e.clientX, e.clientY, window.innerWidth, window.innerHeight);
    mouse.x = p.x; mouse.y = p.y; device = 'keyboard';
  });
  window.addEventListener('mousedown', function () { mouse.down = true; device = 'keyboard'; });
  window.addEventListener('mouseup', function () { mouse.down = false; });

  // playerX/playerY: needed to turn mouse position into an aim direction
  DA.input = {
    state: function (playerX, playerY) {
      var pads = navigator.getGamepads ? navigator.getGamepads() : [];
      var pad = null;
      for (var i = 0; i < pads.length; i++) if (pads[i] && pads[i].connected) { pad = pads[i]; break; }
      if (pad) {
        var gx = DA.applyDeadzone(pad.axes[0]), gy = DA.applyDeadzone(pad.axes[1]);
        var ax = DA.applyDeadzone(pad.axes[2]), ay = DA.applyDeadzone(pad.axes[3]);
        var trigger = pad.buttons[7] && pad.buttons[7].pressed;
        if (gx || gy || ax || ay || trigger) device = 'gamepad';
        if (device === 'gamepad') {
          return { moveX: gx, moveY: gy, aimX: ax, aimY: ay,
                   firing: (ax !== 0 || ay !== 0 || trigger), device: 'gamepad' };
        }
      }
      var mx = (keys.KeyD ? 1 : 0) - (keys.KeyA ? 1 : 0);
      var my = (keys.KeyS ? 1 : 0) - (keys.KeyW ? 1 : 0);
      var aim = DA.norm(mouse.x - playerX, mouse.y - playerY);
      return { moveX: mx, moveY: my, aimX: aim.x, aimY: aim.y,
               firing: mouse.down, device: 'keyboard' };
    }
  };
})();
```

Gamepad notes: right-stick deflection fires (classic twin-stick), right trigger also fires. Standard mapping: axes 0/1 = left stick, 2/3 = right stick, button 7 = right trigger.

**Step 4: Verify tests pass** (✅). Manual: temporarily draw `JSON.stringify(DA.input.state(640,360))` on canvas in `render`; check WASD/mouse and (if available) gamepad respond; remove debug line.

```js
(function () {
  DA.ARENA = { x0: 40, y0: 40, x1: 1240, y1: 680 };
  DA.makePlayer = function () {
    return { x: DA.W / 2, y: DA.H / 2, r: 16, speed: 320,
             hearts: 3, invuln: 0, aimX: 1, aimY: 0, fireCooldown: 0 };
  };
  DA.clampToArena = function (e) {
    e.x = DA.clamp(e.x, DA.ARENA.x0 + e.r, DA.ARENA.x1 - e.r);
    e.y = DA.clamp(e.y, DA.ARENA.y0 + e.r, DA.ARENA.y1 - e.r);
  };
  DA.updatePlayer = function (p, dt) {
    var s = DA.input.state(p.x, p.y);
    var mv = DA.norm(s.moveX, s.moveY);
    p.x += mv.x * p.speed * dt;
    p.y += mv.y * p.speed * dt;
    DA.clampToArena(p);
    if (s.aimX || s.aimY) { p.aimX = s.aimX; p.aimY = s.aimY; }
    p.firing = s.firing;
    if (p.invuln > 0) p.invuln -= dt;
    if (p.fireCooldown > 0) p.fireCooldown -= dt;
  };
  DA.drawPlayer = function (ctx, p) {
    if (p.invuln > 0 && Math.floor(p.invuln * 10) % 2 === 0) return; // blink when hit
    ctx.save();
    ctx.translate(p.x, p.y);
    ctx.rotate(Math.atan2(p.aimY, p.aimX));
    ctx.fillStyle = '#f2f2e9';                       // body
    ctx.beginPath(); ctx.arc(0, 0, p.r, 0, 7); ctx.fill();
    ctx.fillStyle = '#e8d44d';                       // contestant sash
    ctx.fillRect(-p.r, -4, p.r * 2, 8);
    ctx.fillStyle = '#333';                          // gun
    ctx.fillRect(p.r - 4, -3, 14, 6);
    ctx.restore();
  };
})();
```

```js
(function () {
  var SPEED = 700, RATE = 0.14; // seconds between shots
  DA.fireBullet = function (arr, x, y, dx, dy) {
    arr.push({ x: x, y: y, dx: dx, dy: dy, r: 5 });
  };
  DA.updateBullets = function (arr, dt) {
    for (var i = arr.length - 1; i >= 0; i--) {
      var b = arr[i];
      b.x += b.dx * SPEED * dt; b.y += b.dy * SPEED * dt;
      if (b.x < DA.ARENA.x0 || b.x > DA.ARENA.x1 || b.y < DA.ARENA.y0 || b.y > DA.ARENA.y1) arr.splice(i, 1);
    }
  };
  DA.tryPlayerFire = function (p, arr) {
    if (!p.firing || p.fireCooldown > 0) return;
    p.fireCooldown = RATE;
    DA.fireBullet(arr, p.x + p.aimX * 20, p.y + p.aimY * 20, p.aimX, p.aimY);
  };
  DA.drawBullets = function (ctx, arr) {
    ctx.fillStyle = '#ffd94a';
    for (var i = 0; i < arr.length; i++) {
      var b = arr[i];
      ctx.beginPath(); ctx.arc(b.x, b.y, b.r, 0, 7); ctx.fill();
    }
  };
})();
```

```js
(function () {
  var TYPES = {
    shambler: { r: 16, speed: 70,  hp: 2, score: 100, color: '#6fae5c' },
    sprinter: { r: 12, speed: 210, hp: 1, score: 250, color: '#c95d63' }

  // 4 spawn doors: top, bottom, left, right (centered on each wall)
  DA.DOORS = [
    { x: DA.W / 2, y: 20 }, { x: DA.W / 2, y: DA.H - 20 },
    { x: 20, y: DA.H / 2 }, { x: DA.W - 20, y: DA.H / 2 }
  ];
  DA.makeEnemy = function (type, x, y) {
    var t = TYPES[type];
    return { type: type, x: x, y: y, r: t.r, speed: t.speed, hp: t.hp,
             score: t.score, color: t.color, wobble: Math.random() * 6.28 };
  };
  DA.spawnAtDoor = function (arr, type) {
    var d = DA.DOORS[Math.floor(Math.random() * DA.DOORS.length)];
    arr.push(DA.makeEnemy(type, d.x + DA.rand(-30, 30), d.y + DA.rand(-30, 30)));
  };
  DA.updateEnemies = function (arr, player, dt) {
    for (var i = 0; i < arr.length; i++) {
      var e = arr[i];
      var v = DA.norm(player.x - e.x, player.y - e.y);
      e.wobble += dt * 5;
      e.x += (v.x + Math.cos(e.wobble) * 0.25) * e.speed * dt;
      e.y += (v.y + Math.sin(e.wobble) * 0.25) * e.speed * dt;
      DA.clampToArena(e);
    }
  };
  DA.drawEnemies = function (ctx, arr) {
    for (var i = 0; i < arr.length; i++) {
      var e = arr[i];
      ctx.fillStyle = e.color;
      ctx.beginPath(); ctx.arc(e.x, e.y, e.r, 0, 7); ctx.fill();
      ctx.fillStyle = '#1a1a1a'; // dead eyes
      ctx.fillRect(e.x - 6, e.y - 4, 4, 4); ctx.fillRect(e.x + 2, e.y - 4, 4, 4);
    }
  };
})();
```

```js
(function () {
  // A wave = list of spawn groups. interval = seconds between single spawns.
  DA.ROOMS = {
    testRoom: {
      name: 'STUDIO 1',
      waves: [
        [{ type: 'shambler', count: 8,  interval: 0.8 }],
        [{ type: 'shambler', count: 10, interval: 0.6 }, { type: 'sprinter', count: 3, interval: 2.0 }],
        [{ type: 'shambler', count: 14, interval: 0.5 }, { type: 'sprinter', count: 6, interval: 1.2 }]
      ]
    }
  };
  DA.makeWaveManager = function (room) {
    return { room: room, wave: 0, spawners: null, betweenTimer: 2, done: false };
  };
  function startWave(wm) {
    wm.spawners = wm.room.waves[wm.wave].map(function (g) {
      return { type: g.type, left: g.count, interval: g.interval, timer: 0.5 };
    });
  }
  DA.updateWaves = function (wm, enemies, dt) {
    if (wm.done) return;
    if (!wm.spawners) {                   // between waves
      wm.betweenTimer -= dt;
      if (wm.betweenTimer <= 0) startWave(wm);
      return;
    }
    var pending = 0;
    wm.spawners.forEach(function (s) {
      pending += s.left;
      if (s.left <= 0) return;
      s.timer -= dt;
      if (s.timer <= 0) { s.timer = s.interval; s.left--; DA.spawnAtDoor(enemies, s.type); }
    });
    if (pending === 0 && enemies.length === 0) {   // wave cleared
      wm.wave++;
      wm.spawners = null;
      wm.betweenTimer = 2.5;
      if (wm.wave >= wm.room.waves.length) wm.done = true;
    }
  };
})();
```
